#include <stdio.h>

int v(float rs[], int c) {
	float rt = 0.0;
	
	printf("Resistencias individuales:\n");
	for (int i = 0; i < c; i++) {
		printf("Resistencia %d: %.2f ohms\n", i + 1, rs[i]);
		rt += rs[i];
	}
	
	printf("Resistencia total en serie: %.2f ohms\n", rt);
}

int main() {
	float rs[5];
	
	for (int i = 0; i < 5; i++) {
		printf("Ingrese el valor de la resistencia %d: ", i + 1);
		scanf("%f", &rs[i]);
	}
	
	v(rs, 5);
	
	return 0;
}
